# HR-UI v1.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-file HTML dashboard generator that reads a 6-sheet Excel + 1 config sheet and outputs ≤ 50KB offline HTML with 6 modules, 8 viz types, and 3 drill-down points, target: modern desktop + mobile browsers.

**Architecture:** Mapping-driven config (`extractors/mapping.py`) feeds a viz registry (`viz/*.py`) that renders 8 chart types; Alpine.js (inline) handles drill-down and viz switching; single-file output via Python f-string templates.

**Tech Stack:** Python 3.10+ / pandas / openpyxl / Alpine.js 3 (vendored, inline) / Vanilla JS / shanjinki CSS (forked into `vendor/shanjinki.css`).

**Spec:** `docs/superpowers/specs/2026-07-03-hr-ui-v1.2-design.md`

---

- File size: `output/index.html` ≤ 50KB

```python
"""Read .xlsx into dict[module_name, list[dict]]."""
from pathlib import Path
from typing import Any

import openpyxl


def read_workbook(path: Path | str) -> dict[str, list[dict[str, Any]]]:
    """读 .xlsx 全部 sheet（除 [配置] 外）→ dict[sheet_name, records]."""
    wb = openpyxl.load_workbook(path, data_only=True)
    result = {}
    for sheet_name in wb.sheetnames:
        if sheet_name == "[配置]":
            continue
        ws = wb[sheet_name]
        rows = list(ws.iter_rows(values_only=True))
        if not rows:
            result[sheet_name] = []
            continue
        headers = [str(h) if h is not None else f"col_{i}" for i, h in enumerate(rows[0])]
        records = []
        for row in rows[1:]:
            if all(c is None or c == "" for c in row):
                continue  # skip blank rows
            record = {}
            for h, v in zip(headers, row):
                record[h] = v
            records.append(record)
        result[sheet_name] = records
    wb.close()
    return result
```

```python
def read_config(path: Path | str) -> dict[str, Any]:
    """读 [配置] sheet → dict[key, value]."""
    wb = openpyxl.load_workbook(path, data_only=True)
    if "[配置]" not in wb.sheetnames:
        wb.close()
        return {}
    ws = wb["[配置]"]
    config = {}
    for row in ws.iter_rows(min_row=2, values_only=True):
        if row[0] is None or row[0] == "":
            continue
        config[str(row[0])] = row[1] if len(row) > 1 else None
    wb.close()
    return config
```

```python
def clean_value(v: Any, kind: str = "auto") -> Any:
    """清洗 Excel cell 值。

    - kind="pct": "25.86%" → 0.2586；"0.5" → 0.5
    - kind="int": "5,901" → 5901
    - kind="float": "3.14" → 3.14
    - empty/None → None
    """
    if v is None or v == "":
        return None
    if isinstance(v, (int, float)):
        return v
    s = str(v).strip()
    if not s:
        return None
    if kind == "pct":
        if s.endswith("%"):
            return float(s.rstrip("%")) / 100
        return float(s)
    if kind == "int":
        return int(s.replace(",", ""))
    if kind == "float":
        return float(s.replace(",", ""))
    return s
```

```python
from typing import Any

from templates.base import MODULES_ORDER
from viz import VIZ_REGISTRY


def render_section(section: dict, data: Any, options: dict | None = None) -> str:
    """渲染单个 section。"""
    options = options or {}
    viz_type = section["type"]
    alt_types = section.get("alt_types", [])
    drillable = section.get("drillable", False)

    render_fn = VIZ_REGISTRY[viz_type]
    body = render_fn(data, options)

    # viz switch
    switch_html = ""
    if alt_types:
        switch_html = '<div class="viz-switch" style="display:flex;gap:4px;margin-bottom:8px;">'
        switch_html += f'<button class="viz-btn active" data-viz="{viz_type}">{viz_type}</button>'
        for alt in alt_types:
            switch_html += f'<button class="viz-btn" data-viz="{alt}">{alt}</button>'
        switch_html += "</div>"

    # drill button
    drill_html = ""
    if drillable:
        drill_html = f'<button class="drill-btn" data-key="{section["data_key"]}" style="margin-top:8px;padding:8px 16px;min-height:44px;background:#1E5BAA;color:#fff;border:0;border-radius:4px;cursor:pointer;">查看详情</button>'

    return f"""<div class="section" id="sec-{section['key']}">
{('<' + f'h3 style="margin:16px 0 8px 0;">' + section['title'] + '</h3>') if section.get('title') else ''}
{switch_html}
{body}
{drill_html}
</div>"""


def render_module(module_key: str, module_cfg: dict, module_data: dict) -> str:
    """渲染单模块（多个 section 拼装）。"""
    sections_html = []
    for section in module_cfg.get("sections", []):
        data_key = section.get("data_key")
        data = module_data.get(data_key) if data_key else None
        sections_html.append(render_section(section, data))

    return f"""<section class="module" id="M-{module_cfg.get('order', '?')}">
<h2>{module_cfg['icon']} {module_cfg['title']}</h2>
<div class="module-body">
{''.join(sections_html)}
</div>
</section>"""
```

```python
"""钻取数据加载。函数签名: load(key, raw_data) -> dict (JSON-serializable)"""
from typing import Any

from .mapping import MODULES


def _find_section(key: str) -> dict | None:
    for module_cfg in MODULES.values():
        for section in module_cfg.get("sections", []):
            if section.get("data_key") == key:
                return section
    return None


def load_drill_data(key: str, raw_data: dict) -> dict:
    """按 data_key 加载钻取数据。返回 JSON 序列化友好的 dict。"""
    section = _find_section(key)
    if not section:
        return {"error": f"unknown drill key: {key}"}

    rows = raw_data.get(_find_sheet_for_key(key), [])
    return {"key": key, "rows": rows, "count": len(rows)}


def _find_sheet_for_key(key: str) -> str:
    """通过 module 找到对应 sheet 名。简化版：M-1 → 组织架构，以此类推。"""
    mapping = {
        "region_tree": "组织架构",

        "poor_perf_exit": "人员优化",
        "dept_execution": "培训赋能",
```
